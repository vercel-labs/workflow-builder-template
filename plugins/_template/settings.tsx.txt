/**
 * SETTINGS COMPONENT TEMPLATE
 * 
 * This component is displayed in the settings dialog when users configure your integration.
 * It should collect the necessary credentials and configuration.
 */

import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";

export function [IntegrationName]Settings({
  apiKey,
  hasKey,
  onApiKeyChange,
  config,
  onConfigChange,
}: {
  apiKey: string;
  hasKey?: boolean;
  onApiKeyChange: (key: string) => void;
  showCard?: boolean;
  config?: Record<string, string>;
  onConfigChange?: (key: string, value: string) => void;
}) {
  return (
    <div className="space-y-4">
      {/* API Key Field */}
      <div className="space-y-2">
        <Label className="ml-1" htmlFor="[integration-id]ApiKey">
          API Key
        </Label>
        <Input
          className="bg-background"
          id="[integration-id]ApiKey"
          onChange={(e) => onApiKeyChange(e.target.value)}
          placeholder={
            hasKey ? "API key is configured" : "Enter your [Integration Name] API key"
          }
          type="password"
          value={apiKey}
        />
        <p className="text-muted-foreground text-sm">
          Get your API key from{" "}
          <a
            className="text-primary underline"
            href="[LINK_TO_API_KEYS]"
            rel="noopener noreferrer"
            target="_blank"
          >
            [Integration Name] Dashboard
          </a>
          .
        </p>
      </div>

      {/* Add additional config fields as needed */}
      {onConfigChange && (
        <div className="space-y-2">
          <Label className="ml-1" htmlFor="[additional-field]">
            [Additional Field Label]
          </Label>
          <Input
            className="bg-background"
            id="[additional-field]"
            onChange={(e) => onConfigChange("[additional-field-key]", e.target.value)}
            placeholder="[Placeholder]"
            value={config?.["additional-field-key"] || ""}
          />
        </div>
      )}
    </div>
  );
}
